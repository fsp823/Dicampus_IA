document.addEventListener('DOMContentLoaded', () => {
  const billInput = document.getElementById('bill');
  const tipEl = document.getElementById('tip');
  const totalEl = document.getElementById('total');
  const resetBtn = document.getElementById('reset');

  function formatEUR(value) {
    return value.toLocaleString('es-ES', { minimumFractionDigits: 2, maximumFractionDigits: 2 }) + ' €';
  }

  function calculate() {
    const bill = parseFloat(billInput.value);
    if (isNaN(bill) || bill <= 0) {
      tipEl.textContent = formatEUR(0);
      totalEl.textContent = formatEUR(0);
      return;
    }
    const tip = bill * 0.10;
    const total = bill + tip;
    tipEl.textContent = formatEUR(tip);
    totalEl.textContent = formatEUR(total);
  }

  billInput.addEventListener('input', calculate);
  resetBtn.addEventListener('click', () => {
    billInput.value = '';
    calculate();
    billInput.focus();
  });

  calculate();
});

#!/usr/bin/env python3
"""
Calculadora de propinas (CLI) — calcula 10% y muestra total.
Uso: python tip_cli.py
"""

def calcular_propina(bill: float, porcentaje: float = 10.0) -> tuple:
    """Devuelve (propina, total). porcentaje en % (por ejemplo 10.0)."""
    tip = bill * (porcentaje / 100.0)
    total = bill + tip
    return tip, total

def formato_eur(value: float) -> str:
    return f"{value:,.2f} €"

def main():
    try:
        entrada = input("Introduce el valor de la factura (€): ").strip().replace(',', '.')
        bill = float(entrada)
        if bill < 0:
            print("El valor de la factura debe ser mayor o igual a 0.")
            return
    except ValueError:
        print("Entrada no válida. Introduce un número, por ejemplo 42.50")
        return

    porcentaje = 10.0  # fijo al 10%
    tip, total = calcular_propina(bill, porcentaje)

    print(f"Propina ({porcentaje:.0f}%): {formato_eur(tip)}")
    print(f"Total: {formato_eur(total)}")

if __name__ == "__main__":
    main()

    try:
        bill = float(bill_var.get().replace(',', '.'))
    except ValueError:
        bill = 0.0
    try:
        pct = float(pct_var.get().replace(',', '.'))
    except ValueError:
        pct = 10.0

        if people < 1:

# test/test_manual.py
import sys
from pathlib import Path
import math

# Hace importable la carpeta raíz del proyecto (para que 'src' funcione)
ROOT = Path(__file__).resolve().parents[1]
sys.path.insert(0, str(ROOT))

from src.tip_cli import calcular_propina  # asume src/tip_cli.py existe

def safe_calculate(bill, porcentaje=10.0, people=1):
    """
    Valida entradas y llama a calcular_propina.
    - bill: número >= 0
    - porcentaje: número (se permite 0)
    - people: entero >= 1
    Devuelve (tip, total, per_person) o lanza ValueError.
    """
    # Validaciones
    try:
        # convertir y comprobar bill
        bill_f = float(str(bill).replace(',', '.'))
    except Exception as e:
        raise ValueError("Factura no es un número válido") from e
    if bill_f < 0:
        raise ValueError("La factura debe ser un número positivo o cero")

    # porcentaje: convertir a float
    try:
        pct_f = float(str(porcentaje).replace(',', '.'))
    except Exception as e:
        raise ValueError("Porcentaje no es un número válido") from e

    # people: debe ser entero positivo
    try:
        # aceptar strings que representen enteros
        people_i = int(people)
    except Exception as e:
        raise ValueError("Número de personas debe ser un entero") from e
    if people_i < 1:
        raise ValueError("Número de personas debe ser >= 1")

    # Llamada a la función real
    tip, total = calcular_propina(bill_f, pct_f)
    per_person = total / people_i

    return tip, total, per_person

# ---------- Tests (pytest style) ----------

def test_positive_amounts_and_calculation():
    # casos típicos
    cases = [
        (0.0, 10.0, 1),
        (1.23, 10.0, 1),
        (1000.0, 10.0, 4),
    ]
    for bill, pct, people in cases:
        tip, total, per = safe_calculate(bill, pct, people)
        # propina correcta
        assert math.isclose(tip, bill * (pct / 100.0), rel_tol=1e-9)
        # total correcto
        assert math.isclose(total, bill + tip, rel_tol=1e-9)
        # por persona correcto
        assert math.isclose(per, total / people, rel_tol=1e-9)
        # todos los montos no negativos
        assert tip >= 0
        assert total >= 0
        assert per >= 0

def test_negative_bill_raises_value_error():
    try:
        safe_calculate(-5.0, 10.0, 1)
        raised = False
    except ValueError as e:
        raised = True
        assert "factura" in str(e).lower() or "positivo" in str(e).lower()
    assert raised

def test_people_must_be_positive_integer():
    # cero personas
    for invalid in [0, -1, "0", "-3"]:
        try:
            safe_calculate(50.0, 10.0, invalid)
            raised = False
        except ValueError:
            raised = True
        assert raised

    # no entero (float string)
    for invalid in [2.5, "3.7", "dos"]:
        try:
            safe_calculate(50.0, 10.0, invalid)
            raised = False
        except ValueError:
            raised = True
        assert raised

def test_try_except_handling_example():
    # ejemplo de manejo con try/except: la prueba comprueba que el bloque captura el error
    error_caught = False
    try:
        # entrada inválida intencional
        safe_calculate("no-num", 10.0, 1)
    except ValueError as e:
        error_caught = True
        # mensaje útil para el usuario
        assert "no es un número" in str(e).lower() or "factura" in str(e).lower()
    assert error_caught

# Permite ejecutar el archivo directamente sin pytest
if __name__ == "__main__":
    tests = [
        test_positive_amounts_and_calculation,
        test_negative_bill_raises_value_error,

    for t in tests:
        try:
            t()
            print(f"{t.__name__}: OK")
        except AssertionError as ae:
            print(f"{t.__name__}: FALLÓ -> {ae}")
        except Exception as ex:
            print(f"{t.__name__}: ERROR inesperado -> {ex}")

# src/tip_cli.py
from typing import Tuple

def calcular_propina(bill: float, porcentaje: float = 10.0) -> Tuple[float, float]:
    """
    Calcula la propina y el total.
    - bill: importe de la factura (>= 0)
    - porcentaje: porcentaje de propina (puede ser 0 o positivo)
    Devuelve (propina, total). Lanza ValueError en entradas inválidas.
    """
    try:
        bill_f = float(str(bill).replace(',', '.'))
    except Exception as e:
        raise ValueError("Factura no es un número válido") from e
    if bill_f < 0:
        raise ValueError("La factura debe ser un número positivo o cero")

    try:
        pct_f = float(str(porcentaje).replace(',', '.'))
    except Exception as e:
        raise ValueError("Porcentaje no es un número válido") from e
    if pct_f < 0:
        raise ValueError("El porcentaje debe ser >= 0")

    tip = bill_f * (pct_f / 100.0)
    total = bill_f + tip
    return tip, total

from typing import Tuple

def calcular_propina(bill: float, porcentaje: float = 10.0, people: int = 1) -> Tuple[float, float, float]:
    """
    Calcula la propina, el total y el importe por persona.

    Parámetros
    - bill: importe de la factura (>= 0). Se aceptan strings con coma o punto.
    - porcentaje: porcentaje de propina (>= 0). Se aceptan strings con coma o punto.
    - people: número de personas (entero >= 1). Se aceptan strings que representen enteros.

    Devuelve
    - (tip, total, per_person)

    Lanza ValueError con mensajes claros si alguna entrada es inválida.
    """
    # Validar y convertir bill
    try:
        bill_f = float(str(bill).replace(',', '.'))
    except Exception as e:
        raise ValueError("Factura no es un número válido") from e
    if bill_f < 0:
        raise ValueError("La factura debe ser un número positivo o cero")

    # Validar y convertir porcentaje
    try:
        pct_f = float(str(porcentaje).replace(',', '.'))
    except Exception as e:
        raise ValueError("Porcentaje no es un número válido") from e
    if pct_f < 0:
        raise ValueError("El porcentaje debe ser >= 0")

    # Validar y convertir people
    try:
        people_i = int(str(people))
    except Exception as e:
        raise ValueError("Número de personas debe ser un entero") from e
    if people_i < 1:
        raise ValueError("Número de personas debe ser >= 1")

    tip = bill_f * (pct_f / 100.0)
    total = bill_f + tip
    per_person = total / people_i

    return tip, total, per_person

donde debo colocar el gitignore para quitar archivos en github
Dónde colocar .gitignore y cómo funciona
Coloca un archivo llamado .gitignore en la raíz del repositorio (la carpeta que contiene .git/). Ese fichero controla qué archivos o patrones Git ignora para todo el repositorio y es la forma más habitual de “quitar” archivos de futuros commits en GitHub.

Opciones y alcance de los .gitignore
.gitignore en la raíz del repositorio — aplica a todo el repositorio; recomendado para la mayoría de los casos.

.gitignore por subcarpeta — puedes tener varios .gitignore en subdirectorios; cada uno aplica a su carpeta y subcarpetas.

Ignorar globalmente en tu máquina — para reglas personales (por ejemplo *.log, .DS_Store) usa un archivo global, p. ej. ~/.gitignore_global, y configúralo con git config --global core.excludesfile ~/.gitignore_global.

Ejemplo mínimo de .gitignore en la raíz
gitignore
# Archivos de entorno y dependencias
venv/
__pycache__/
*.pyc

# Archivos de sistema
.DS_Store
Thumbs.db

# Archivos de editor
.vscode/
.idea/

# Archivos de build y binarios
dist/
build/
*.exe
Guarda ese fichero como .gitignore en la raíz y haz commit.

Qué hacer si los archivos ya están en el repositorio (ya fueron commiteados)
Si un archivo ya está trackeado por Git, añadirlo al .gitignore no lo elimina del historial ni deja de trackearlo automáticamente. Para dejar de trackearlo sin borrarlo localmente:

Desde la raíz del repo:

bash
git rm --cached ruta/al/archivo
git commit -m "Dejar de trackear archivo X"
git push
Para dejar de trackear todos los archivos que ahora coinciden con .gitignore:

bash
git rm -r --cached .
git add .
git commit -m "Aplicar .gitignore y dejar de trackear archivos ignorados"
git push
Nota: git rm --cached quita el archivo del índice (staging) y del próximo commit, pero no borra el archivo en tu disco local.

Si necesitas eliminar archivos sensibles del historial (más complejo)
Para borrar archivos de todos los commits anteriores (por ejemplo, claves o ficheros grandes), hay que reescribir el historial. Herramientas comunes: BFG Repo-Cleaner o git filter-repo.

Reescribir historial cambia los hashes de commits; todos los colaboradores deberán volver a clonar o forzar sincronizar. Es una operación con impacto, planifícala y haz backup antes.

Comprobaciones útiles después de aplicar .gitignore
Verifica qué archivos están siendo ignorados:

bash
git status --ignored
Verifica si un archivo concreto está siendo ignorado:

bash
git check-ignore -v ruta/al/archivo

¿cómo redondeo números a dos decimales en Python?
Redondeo a dos decimales — respuesta rápida
Usa round() para cálculos rápidos y format/f-strings para mostrar; para dinero o precisión exacta usa decimal.Decimal con quantize.

Ejemplos prácticos
1. round() (resultado numérico)
python
x = 2.34567
y = round(x, 2)   # 2.35
2. Formateo para mostrar (cadena)
python
x = 2.34567
s1 = "{:.2f}".format(x)     # '2.35'
s2 = f"{x:.2f}"             # '2.35'
s3 = format(x, ".2f")       # '2.35'
3. decimal.Decimal para control de redondeo (recomendado para dinero)
python
from decimal import Decimal, ROUND_HALF_UP

x = Decimal("2.345")
rounded = x.quantize(Decimal("0.00"), rounding=ROUND_HALF_UP)  # Decimal('2.35')
4. Forzar truncado (sin redondeo)
python
import math

x = 2.349
trunc2 = math.floor(x * 100) / 100  # 2.34
Diferencias importantes y cuándo usar cada uno
round() devuelve un float. Es rápido pero puede mostrar artefactos por la representación binaria de los float (por ejemplo round(2.675, 2) puede dar 2.67 en algunas versiones de Python).

Formateo ("{:.2f}" o f-strings) es ideal para presentar valores al usuario; devuelve una cadena con exactamente dos decimales.

decimal.Decimal es la opción correcta para finanzas o cuando necesitas un comportamiento de redondeo determinista; permite elegir modos de redondeo (ROUND_HALF_UP, ROUND_HALF_EVEN, etc.).

Truncado con math.floor sirve si necesitas cortar sin redondear.

Modos de redondeo comunes con decimal
ROUND_HALF_UP — redondea .5 hacia arriba (comportamiento humano esperado).

ROUND_HALF_EVEN — redondeo bancario (empata a par).
Ejemplo:

python
from decimal import Decimal, ROUND_HALF_EVEN, ROUND_HALF_UP

x = Decimal("2.5")
x.quantize(Decimal("0"), rounding=ROUND_HALF_UP)    # 3
x.quantize(Decimal("0"), rounding=ROUND_HALF_EVEN)  # 2
Buenas prácticas para aplicaciones reales
Para cálculos monetarios usa Decimal desde el inicio (no conviertas desde float).

Para mostrar valores al usuario, formatea con f"{value:.2f}" o con locale si necesitas separadores y símbolo local.

Evita confiar en float para comparaciones de igualdad; usa tolerancias (math.isclose) o Decimal.

Comprueba casos límite: 2.675, 2.5, -1.235.

Verifica el modo de redondeo que necesitas (humano vs bancario).

Si usas Decimal, añade tests que confirmen el quantize con el rounding elegido.

# src/tip_cli.py
from typing import Tuple
from decimal import Decimal, ROUND_HALF_UP, InvalidOperation

def _to_decimal(value, name: str) -> Decimal:
    try:
        s = str(value).strip().replace(',', '.')
        return Decimal(s)
    except (InvalidOperation, ValueError, TypeError) as e:
        raise ValueError(f"{name} no es un número válido") from e

def calcular_propina(bill, porcentaje: float = 10.0, people: int = 1) -> Tuple[float, float, float]:
    """
    Calcula la propina, el total y el importe por persona usando redondeo humano (ROUND_HALF_UP).

    - bill: importe de la factura (>= 0). Se aceptan números o strings con coma o punto.
    - porcentaje: porcentaje de propina (>= 0). Se aceptan números o strings con coma o punto.
    - people: número de personas (entero >= 1). Se aceptan enteros o strings que representen enteros.

    Devuelve (tip, total, per_person) como floats redondeados a 2 decimales.
    Lanza ValueError con mensajes claros si alguna entrada es inválida.
    """
    # Convertir y validar bill
    bill_d = _to_decimal(bill, "Factura")
    if bill_d < 0:
        raise ValueError("La factura debe ser un número positivo o cero")

    # Convertir y validar porcentaje
    pct_d = _to_decimal(porcentaje, "Porcentaje")
    if pct_d < 0:
        raise ValueError("El porcentaje debe ser >= 0")

    # Validar people como entero
    try:
        people_i = int(str(people))
    except Exception as e:
        raise ValueError("Número de personas debe ser un entero") from e
    if people_i < 1:
        raise ValueError("Número de personas debe ser >= 1")

    # Cálculo con Decimal para evitar errores de float
    # tip = bill * (pct / 100)
    tip_d = (bill_d * (pct_d / Decimal('100'))).quantize(Decimal('0.00'), rounding=ROUND_HALF_UP)
    total_d = (bill_d + tip_d).quantize(Decimal('0.00'), rounding=ROUND_HALF_UP)
    per_person_d = (total_d / Decimal(people_i)).quantize(Decimal('0.00'), rounding=ROUND_HALF_UP)

    # Devolver como floats (compatibilidad con tests existentes)
    return float(tip_d), float(total_d), float(per_person_d)
Notas breves
Redondeo humano: usa ROUND_HALF_UP vía decimal.Decimal.quantize para comportarse como el redondeo que la mayoría espera (0.5 hacia arriba).

Entradas: acepta strings con coma decimal (p. ej. "1,23") y números; lanza ValueError con mensajes consistentes con las pruebas.

Salida: devuelve float para mantener compatibilidad con las pruebas unittest existentes; internamente se calcula y redondea con Decimal.

¿Cómo formateo un número como moneda con el símbolo $?
Formatear números como moneda con el símbolo dólar
A continuación tienes métodos prácticos (desde el más simple al más robusto), ejemplos listos para copiar y notas sobre cuándo usar cada uno.

1. Formateo rápido con f-strings o format
Útil para mostrar valores con separador de miles en estilo inglés y dos decimales.

No aplica reglas locales (p. ej. separador decimal distinto).

python
value = 12345.678
s = f"${value:,.2f}"      # -> '$12,345.68'
# o
s2 = "${:,.2f}".format(value)
Ventaja: simple y sin dependencias.
Limitación: siempre usa coma para miles y punto para decimales.

2. Usar locale para formato local y símbolo
Útil si quieres respetar convenciones regionales (separadores, posición del símbolo).

Requiere configurar la locale del sistema; en Windows los nombres pueden variar.

python
import locale

# Ejemplo para Estados Unidos (Linux/macOS suele aceptar 'en_US.UTF-8')
locale.setlocale(locale.LC_ALL, 'en_US.UTF-8')
value = 12345.678
s = locale.currency(value, symbol=True, grouping=True)  # -> '$12,345.68'
Notas:

En Windows puede ser 'English_United States.1252'.

locale.setlocale afecta todo el proceso; ten cuidado en aplicaciones multihilo.

Si setlocale falla, prueba con otras variantes o usa babel.

3. Usar babel para formateo robusto por locale (recomendado para apps)
babel maneja locales sin cambiar el estado global y es consistente entre plataformas.

Instalar: pip install Babel

python
from babel.numbers import format_currency

value = 12345.678
s = format_currency(value, 'USD', locale='en_US')   # -> '$12,345.68'
# Para español de España con símbolo $ (raro, pero posible)
s2 = format_currency(value, 'USD', locale='es_ES')  # -> '12.345,68 US$'
Ventaja: control fino, consistente y seguro en servidores.
Limitación: dependencia externa.

4. Formateo con Decimal para dinero y luego mostrar con $
Usa decimal.Decimal para cálculos exactos y ROUND_HALF_UP para redondeo humano, luego formatea la cadena.

python
from decimal import Decimal, ROUND_HALF_UP

value = Decimal('12345.678')
rounded = value.quantize(Decimal('0.00'), rounding=ROUND_HALF_UP)  # Decimal('12345.68')
s = f"${rounded:,.2f}"  # -> '$12,345.68'  (rounded es Decimal, formateo funciona)
Ventaja: precisión en cálculos monetarios; control de redondeo.

5. Ejemplo completo integrable en tu calculadora
Devuelve cadena con $ y formato legible, usando Decimal internamente:

python
from decimal import Decimal, ROUND_HALF_UP

def format_usd(amount):
    d = Decimal(str(amount)).quantize(Decimal('0.00'), rounding=ROUND_HALF_UP)
    # convertir a float para usar formateo con separadores
    return f"${float(d):,.2f}"

# Uso
print(format_usd('12345.678'))  # -> $12,345.68
Consideraciones y recomendaciones
Para scripts rápidos usa f-strings (f"${value:,.2f}").

Para aplicaciones que respeten locales usa locale o, mejor, babel.

Para dinero y cálculos usa Decimal y ROUND_HALF_UP antes de formatear.

Posición del símbolo y separadores dependen del locale; babel es la opción más segura si tu app atiende usuarios internacionales.

Evita cambiar globalmente la locale en aplicaciones multihilo o servidores; prefiere babel o formateo local por llamada.

from typing import Tuple
from decimal import Decimal, ROUND_HALF_UP, InvalidOperation
import locale

def _to_decimal(value, name: str) -> Decimal:
    try:
        s = str(value).strip().replace(',', '.')
        return Decimal(s)
    except (InvalidOperation, ValueError, TypeError) as e:
        raise ValueError(f"{name} no es un número válido") from e

def calcular_propina(bill, porcentaje: float = 10.0, people: int = 1) -> Tuple[float, float, float]:
    bill_d = _to_decimal(bill, "Factura")
    if bill_d < 0:
        raise ValueError("La factura debe ser un número positivo o cero")

    pct_d = _to_decimal(porcentaje, "Porcentaje")
    if pct_d < 0:
        raise ValueError("El porcentaje debe ser >= 0")

    try:
        people_i = int(str(people))
    except Exception as e:
        raise ValueError("Número de personas debe ser un entero") from e
    if people_i < 1:
        raise ValueError("Número de personas debe ser >= 1")

    tip_d = (bill_d * (pct_d / Decimal('100'))).quantize(Decimal('0.00'), rounding=ROUND_HALF_UP)
    total_d = (bill_d + tip_d).quantize(Decimal('0.00'), rounding=ROUND_HALF_UP)
    per_person_d = (total_d / Decimal(people_i)).quantize(Decimal('0.00'), rounding=ROUND_HALF_UP)

    return float(tip_d), float(total_d), float(per_person_d)

# -------------------------
# Formateo de moneda local
# -------------------------
def _try_babel_format(amount, currency_code, locale_str):
    try:
        from babel.numbers import format_currency
    except Exception:
        return None
    try:
        return format_currency(amount, currency_code, locale=locale_str)
    except Exception:
        return None

def _locale_currency_fallback(amount):
    """
    Usa locale.currency tras configurar locale al valor por defecto del sistema.
    Si falla, devuelve formato simple con símbolo USD.
    """
    try:
        # '' indica usar la configuración regional del entorno
        locale.setlocale(locale.LC_ALL, '')
    except Exception:
        # no podemos cambiar locale; seguir con la actual
        pass
    try:
        # locale.currency maneja agrupación si grouping=True
        return locale.currency(amount, symbol=True, grouping=True)
    except Exception:
        # fallback simple: símbolo dólar con separador de miles en estilo inglés
        return f"${amount:,.2f}"

def format_currency_local(amount) -> str:
    """
    Formatea amount como moneda según la configuración regional del sistema.
    - Intenta usar babel si está disponible (más robusto).
    - Si no, usa locale.currency.
    - Si todo falla, devuelve '$12,345.68' como fallback.
    Acepta Decimal, float o string.
    """
    # convertir a Decimal y redondear humanamente
    try:
        d = _to_decimal(amount, "Cantidad")
    except ValueError as e:
        raise ValueError("Cantidad no es un número válido") from e
    d_q = d.quantize(Decimal('0.00'), rounding=ROUND_HALF_UP)

    # intentar detectar locale y currency
    # 1) obtener locale del sistema
    loc = None
    try:
        loc = locale.getdefaultlocale()[0]  # e.g., 'es_ES'
    except Exception:
        loc = None

    # 2) si babel está disponible, intentar formatear con la moneda local detectada
    if loc:
        # intentar inferir moneda por locale (ej. 'es_ES' -> 'EUR', 'en_US' -> 'USD')
        # babel puede formatear con el código de moneda; si no conocemos el código, usar 'USD' como fallback
        # intentamos con 'USD' y 'EUR' según locale prefix
        currency_candidates = []
        if loc.lower().startswith('es') or loc.lower().startswith('fr') or loc.lower().startswith('de'):
            currency_candidates = ['EUR', 'USD']
        elif loc.lower().startswith('en_us') or loc.lower().startswith('en'):
            currency_candidates = ['USD', 'EUR']
        else:
            currency_candidates = ['USD', 'EUR']

        for code in currency_candidates:
            s = _try_babel_format(float(d_q), code, loc)
            if s:
                return s

    # 3) fallback a locale.currency
    try:
        return _locale_currency_fallback(float(d_q))
    except Exception:
        return f"${float(d_q):,.2f}"

from typing import Tuple, Optional
from decimal import Decimal, ROUND_HALF_UP, InvalidOperation
import locale

def _to_decimal(value, name: str) -> Decimal:
    try:
        s = str(value).strip().replace(',', '.')
        return Decimal(s)
    except (InvalidOperation, ValueError, TypeError) as e:
        raise ValueError(f"{name} no es un número válido") from e

def calcular_propina(bill, porcentaje: float = 10.0, people: int = 1, fixed_tip: Optional[float] = None) -> Tuple[float, float, float]:
    """
    Calcula la propina, el total y el importe por persona.

    Parámetros
    - bill: importe de la factura (>= 0). Se aceptan números o strings con coma o punto.
    - porcentaje: porcentaje de propina (>= 0). Se aceptan números o strings con coma o punto.
    - people: número de personas (entero >= 1). Se aceptan enteros o strings que representen enteros.
    - fixed_tip: si se proporciona (no None), se usa este monto fijo de propina en la misma moneda
                 en lugar de calcularla por porcentaje. Se aceptan números o strings.

    Devuelve (tip, total, per_person) como floats redondeados a 2 decimales.
    Lanza ValueError con mensajes claros si alguna entrada es inválida.
    """
    bill_d = _to_decimal(bill, "Factura")
    if bill_d < 0:
        raise ValueError("La factura debe ser un número positivo o cero")

    # Validar people como entero
    try:
        people_i = int(str(people))
    except Exception as e:
        raise ValueError("Número de personas debe ser un entero") from e
    if people_i < 1:
        raise ValueError("Número de personas debe ser >= 1")

    # Si fixed_tip se proporciona, usarlo (validar)
    if fixed_tip is not None:
        fixed_d = _to_decimal(fixed_tip, "Propina fija")
        if fixed_d < 0:
            raise ValueError("La propina fija debe ser >= 0")
        tip_d = fixed_d.quantize(Decimal('0.00'), rounding=ROUND_HALF_UP)
    else:
        pct_d = _to_decimal(porcentaje, "Porcentaje")
        if pct_d < 0:
            raise ValueError("El porcentaje debe ser >= 0")
        tip_d = (bill_d * (pct_d / Decimal('100'))).quantize(Decimal('0.00'), rounding=ROUND_HALF_UP)

    total_d = (bill_d + tip_d).quantize(Decimal('0.00'), rounding=ROUND_HALF_UP)
    per_person_d = (total_d / Decimal(people_i)).quantize(Decimal('0.00'), rounding=ROUND_HALF_UP)

    return float(tip_d), float(total_d), float(per_person_d)

def _try_babel_format(amount, currency_code, locale_str):
    try:
        from babel.numbers import format_currency
    except Exception:
        return None
    try:
        return format_currency(amount, currency_code, locale=locale_str)
    except Exception:
        return None

def _locale_currency_fallback(amount):
    try:
        locale.setlocale(locale.LC_ALL, '')
    except Exception:
        pass
    try:
        return locale.currency(amount, symbol=True, grouping=True)
    except Exception:
        return f"${amount:,.2f}"

def format_currency_local(amount) -> str:
    try:
        d = _to_decimal(amount, "Cantidad")
    except ValueError as e:
        raise ValueError("Cantidad no es un número válido") from e
    d_q = d.quantize(Decimal('0.00'), rounding=ROUND_HALF_UP)

    loc = None
    try:
        loc = locale.getdefaultlocale()[0]
    except Exception:
        loc = None

    if loc:
        currency_candidates = []
        if loc.lower().startswith(('es', 'fr', 'de')):
            currency_candidates = ['EUR', 'USD']
        elif loc.lower().startswith('en_us') or loc.lower().startswith('en'):
            currency_candidates = ['USD', 'EUR']
        else:
            currency_candidates = ['USD', 'EUR']

        for code in currency_candidates:
            s = _try_babel_format(float(d_q), code, loc)
            if s:
                return s

    try:
        return _locale_currency_fallback(float(d_q))
    except Exception:
        return f"${float(d_q):,.2f}"

Qué debe contener un buen docstring
Resumen breve — una línea que describa qué hace la función (verbo en presente).

Descripción ampliada — si hace falta, uno o dos párrafos que expliquen el comportamiento, efectos secundarios y restricciones.

Parámetros — nombre, tipo esperado y breve explicación de cada argumento; indicar valores por defecto y formatos aceptados.

Valor devuelto — tipo y significado del valor retornado; estructura si devuelve tuplas o colecciones.

Excepciones — qué errores puede lanzar y en qué condiciones.

Ejemplos de uso — fragmentos mínimos que muestren llamadas típicas y resultados esperados.

Notas adicionales — advertencias, complejidad algorítmica, referencias a especificaciones o a otras funciones relacionadas.

def dividir(numerador, denominador):
    """
    Divide numerador entre denominador y devuelve el cociente.

    Parámetros
    numerador (int | float) -- valor a dividir.
    denominador (int | float) -- valor divisor; no puede ser 0.

    Devuelve
    float -- resultado de la división.

Estilos comunes y cuándo elegirlos
One-line docstring — para funciones triviales; debe ser una sola frase clara.

PEP 257 / triple-quote — convención base de Python; buena para compatibilidad con herramientas.

Google style — legible y compacto; popular en proyectos Google-like.

NumPy style — ideal para funciones científicas con secciones de Parameters, Returns, Examples y tipos más detallados.

Tradeoffs: Google y NumPy son más verbosos pero facilitan la generación automática de documentación; la one-line es concisa pero insuficiente para funciones no triviales.

Buenas prácticas y recomendaciones concretas
Empieza con un verbo en presente: Calcula, Devuelve, Abre.

Sé explícito con tipos (acepta str con coma decimal, int >= 1, etc.).

Documenta efectos secundarios (modifica archivo, hace I/O, cambia estado global).

Incluye ejemplos ejecutables (doctest) si quieres que sean comprobables automáticamente.

Mantén el docstring sincronizado con la firma: actualízalo cuando cambie la API.

No repitas el nombre de la función en la primera línea (evita: "función foo hace...").

Limita la longitud: la primera línea no debe exceder ~72 caracteres para que herramientas la muestren bien.

Usa Raises o Excepciones para que los usuarios sepan qué capturar.

Evita detalles de implementación que puedan cambiar; documenta la contrato y el comportamiento observable.

Errores comunes a evitar
Docstrings vacíos o inexistentes en funciones públicas.

Documentar sólo tipos vagos como object o any sin contexto.

Poner ejemplos que no funcionan (cambian API o dependen de estado externo).

Dejar mensajes ambiguos en las secciones de errores (p. ej., “lanza Exception” sin especificar cuál).

Herramientas y automatización útil
Sphinx o pdoc para generar documentación HTML a partir de docstrings.

doctest para ejecutar ejemplos incluidos en docstrings.

Linters (p. ej., pylint, flake8 con plugins) para detectar docstrings faltantes o mal formateados.

Plantillas de estilo (Google/NumPy) integradas en generadores de docs para consistencia.

        total_var.set(format_currency_local(total))
        per_var.set(format_currency_local(per_person))
    except ValueError as ve:
        tip_var.set("—")
        total_var.set("—")
        per_var.set("—")
        error_var.set(str(ve))
    except Exception as e:
        tip_var.set("—")
        total_var.set("—")
        per_var.set("—")
        error_var.set("Error inesperado: " + str(e))

def reset(*_):
    bill_var.set("0.00")
    pct_var.set("10")
    fixed_var.set("0.00")
    people_var.set("1")
    mode_var.set('pct')
    actualizar()
